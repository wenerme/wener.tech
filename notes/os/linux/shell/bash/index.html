# you can also put all your commands into a fileecho "ls\nls" > somefilecat somefile | parallel
# finally, just straight command line parametersparallel echo ::: a b c
# by default, it runs as many jobs in parallel as there are cores in your computer
# if you try and set more jobs than there are cores, they will be concurrent, but# they will only ever be parallel up to your core count
# for httpie use the --ignore-stdin flag, or else it gets mixed upseq 1 | parallel -n0 "http --ignore-stdin POST http://httpbin.org/post url=http://google.com/"
