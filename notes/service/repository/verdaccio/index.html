FROM verdaccio/verdaccio:4USER rootENV NODE_ENV=productionRUN yarn global add verdaccio-minio && \  ln -s /usr/local/share/.config/yarn/global/node_modules/verdaccio-minio /verdaccio/plugins/verdaccio-minio  && \  chown -R 10001 /usr/local/share/.config/yarn/global/node_modules/verdaccio-minio && \  chown -R 10001 /verdaccio/plugins
USER verdaccio
