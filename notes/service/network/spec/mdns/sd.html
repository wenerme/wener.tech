mDNS 服务定义

  • _services._dns-sd._udp
  • Linux avahi
  • 参考
    • rfc6763 DNS-Based Service Discovery
    • Bonjour Printing Specification
    • http://www.dns-sd.org/

macOS​

# 暴露服务
dns-sd -R "My test server with metrics-endpoint" _prometheus-http._tcp. . 9000 path=/metrics
# 扫描
dns-sd -B
# 所有服务
dns-sd -B _services._dns-sd._udp local.
# 查找打印机
dns-sd -B _ipp._tcp local.
# 查看内容里的 adminurl
dns-sd -Z _ipp._tcp local.
# 获取 IP
dns-sd -Gv4v6 LenovoAB66CD.local.
dns-sd -Gv4v6 LenovoAD69CD.local.

# 域名 zone 格式
# PTR SRV TXT
dns-sd -Z
