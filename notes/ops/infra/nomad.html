Nomad

  • 是什么？
    • 分布式调度器
    • 工作负载编排
  • 核心卖点： 混合工作负载编排
  • 特点
    • Agent 方式运行
    • UI 提供基本信息和操作
    • 声明式任务定义
    • 主要调度 工作负载/Workload - 不包含网络、存储、服务、配置等
    • 支持 CSI 存储插件
  • Driver
    • Docker
    • Fork/Exec
    • Java
    • Podman
    • Qemu
  • 社区 Driver
    • Containerd
    • Firecracker
    • Jailtask
    • LXC
    • Pot
    • Sigularity
    • Systemd nspawn
    • Windows IIS
  • 端口
    • 4646/HTTP
    • 4647/RPC
    • 4648/Serf
  • 集成
    • consul - 服务发现、注册、健康检查
    • consul connect - 网络
    • vault - 密钥
  • hashicorp/go-getter - 获取 artifact
适用场景
  • Windows, macOS 调度
  • 非标准资源调度 - GPU、外部设备
  • 非标准 CPU arch 调度

cat <<HCL > server.hcl
# Increase log verbosity
log_level = "DEBUG"

# Setup data dir
data_dir = "/tmp/server1"

# Enable the server
server {
    enabled = true

    # Self-elect, should be 3 or 5 for production
    bootstrap_expect = 1
}
HCL
nomad agent -config server.hcl
# Increase log verbosity
log_level = "DEBUG"

# Setup data dir
data_dir = "/tmp/client1"

# Give the agent a unique name. Defaults to hostname
name = "client1"

# Enable the client
client {
    enabled = true

    # For demo assume we are talking to server1. For production,
    # this should be like "nomad.service.consul:4647" and a system
    # like Consul used for service discovery.
    servers = ["127.0.0.1:4647"]
}

}

  enabled          = true
  bootstrap_expect = 3
  data_dir = "/opt/nomad/server"

  enabled_schedulers = ["batch", "service"]
  num_schedulers     = 7

    retry_join     = [ "1.1.1.1", "2.2.2.2" ]
    retry_max      = 3

    scheduler_algorithm = "spread"

      batch_scheduler_enabled   = true
      system_scheduler_enabled  = true

  }
}
