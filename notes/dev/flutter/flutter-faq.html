PREFIX=${PREFIX:-"Icon-App"}
source=${1:-logo.svg}
dst=${2:-.}
for s in $SIZES;do
  size=${s%x*}
  scale=${s##*x}
  resize=$(bc <<< ${size}*${scale} )
  out=$dst/"$PREFIX-${size}x${size}@${scale}x.png"
  echo Generate $out
  convert $CONVERT_EXTRA "$source" -resize ${resize}x${resize}     -unsharp 1x4 $out
done
